create table followers
(
	id int(10) unsigned not null auto_increment
		primary key,
	user_id int(10) unsigned not null,
	follower_id int(10) unsigned not null
)
;

create table login_tokens
(
	id int not null auto_increment
		primary key,
	token char(64) default '' not null,
	user_id int not null,
	constraint token
		unique (token)
)
;

create index user_id
	on login_tokens (user_id)
;

create table password_tokens
(
	id int not null auto_increment
		primary key,
	token char(64) default '' not null,
	user_id int not null,
	constraint token
		unique (token)
)
;

create table relationship
(
	user_one int(10) unsigned not null,
	user_two int(10) unsigned not null,
	status tinyint not null,
	actionUserID int(10) unsigned not null
)
;

create index relationship_users_id_id_id_fk
	on relationship (user_one, user_two, actionUserID)
;

create table users
(
	id int not null auto_increment
		primary key,
	username varchar(32) null,
	password varchar(60) null,
	email text null,
	join_date date not null
)
;

alter table login_tokens
	add constraint login_tokens_ibfk_1
		foreign key (user_id) references users (id)
;
